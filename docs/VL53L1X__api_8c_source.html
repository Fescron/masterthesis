225 VL53L1X_ERROR VL53L1X_SensorInit(uint16_t dev)
226 {
227  VL53L1X_ERROR status = 0;
228  uint8_t Addr = 0x00, tmp;
229 
230  for (Addr = 0x2D; Addr <= 0x87; Addr++){
231  status = VL53L1_WrByte(dev, Addr, VL51L1X_DEFAULT_CONFIGURATION[Addr - 0x2D]);
232  }
233  status = VL53L1X_StartRanging(dev);
234  tmp = 0;
235  while(tmp==0){
236  status = VL53L1X_CheckForDataReady(dev, &tmp);
237  }
238  status = VL53L1X_ClearInterrupt(dev);
239  status = VL53L1X_StopRanging(dev);
240  status = VL53L1_WrByte(dev, VL53L1_VHV_CONFIG__TIMEOUT_MACROP_LOOP_BOUND, 0x09); /* two bounds VHV */
241  status = VL53L1_WrByte(dev, 0x0B, 0); /* start VHV from the previous temperature */
242  return status;
243 }
