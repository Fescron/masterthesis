70 /**************************************************************************//**
71  * @brief
72  * Soft reset the SHT30.
73  *
74  * @details
75  * This method also initializes i2C if necessary.
76  *
77  * @note
78  * This is a static method because it's only internally used in this file
79  * and called by other methods if necessary.
80  *
81  * @return
82  * @return
83  * @li `0` - Byte written correctly.
84  * @li `-1` - `i2cTransferNack`, NACK received during transfer.
85  * @li `-2` - `i2cTransferBusErr`, Bus error during transfer (misplaced START/STOP).
86  * @li `-3` - `i2cTransferArbLost`, Arbitration lost during transfer.
87  * @li `-4` - `i2cTransferUsageFault`, Usage fault.
88  * @li `-5` - `i2cTransferSwFault`, SW fault.
89  * @li `-6` - Coudldn't configure I2C.
90  *****************************************************************************/
91 int8_t softResetSHT30 (void)
92 {
93  /* Initialize I2C if not already the case */
94  if (initI2C() != 0) return (-6);
95 
96  /* Separate command in two bytes (ANDing with a mask and shifting one byte (8 bits) per position) */
97  uint8_t cmd_buf[2] = {(SOFT_RESET_CMD >> 8), (SOFT_RESET_CMD & 0x00FF)};
98 
99  /* Transfer structure */
100  I2C_TransferSeq_TypeDef i2cTransfer;
101  I2C_TransferReturn_TypeDef result;
102 
103  /* Initializing I2C transfer */
104  i2cTransfer.addr = SHT30_ADDRESS << 1; /* Left shift so we go to 8 bit notation instead of 7 */
105  i2cTransfer.flags = I2C_FLAG_WRITE; /* Data written from buf[0].data */
106  i2cTransfer.buf[0].data = cmd_buf;
107  i2cTransfer.buf[0].len = 2; /* Command is two bytes */
108  result = I2C_TransferInit(BUS_I2C, &i2cTransfer);
109 
110  /* Transmitting I2C data (busy-wait until transfer is complete) */
111  while (result == i2cTransferInProgress) result = I2C_Transfer(BUS_I2C);
112 
113 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
114  dbinfo("Soft reset SHT30");
115 #endif /* DEBUG_DBPRINT */
116 
117  /* Indicate that SHT30 has been soft reset */
118  SHT30_reset = true;
119 
120  return (result);
121 }
