56 /**************************************************************************//**
57  * @brief
58  * Initialize the sensor.
59  *
60  * @details
61  * This method wait until the sensor is booted, initializes it and
62  * gets offset values.
63  *
64  * @return
65  * @return
66  * @li `0` - Calibration successful.
67  * @li `-1` - Calibration failed.
68  *****************************************************************************/
69 int8_t initVL53L1X (void)
70 {
71  /* Returned status */
72  int8_t status;
73 
74  /* Wait for device booted */
75  uint8_t bootState = 0;
76  uint8_t exitWhile = 0;
77 
78  while ((bootState != 1) && (exitWhile < 5))
79  {
80  status = VL53L1X_BootState(0x52, &bootState);
81 
82 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
83  if (status != 0) dbcritInt("VL53L1X_BootState status = ", status, "");
84  dbinfoInt("VL53L1X_BootState = ", bootState, "");
85 #endif /* DEBUG_DBPRINT */
86 
87  /* 1s delay */
88  delay(1000);
89 
90  /* Functionality so we don't end up forever in this loop */
91  exitWhile++;
92  }
93 
94  /* Initialize sensor */
95  status = VL53L1X_SensorInit(0x52);
96 
97 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
98  if (status != 0) dbcritInt("VL53L1X_SensorInit status = ", status, "");
99 #endif /* DEBUG_DBPRINT */
100 
101  /* Get offset */
102  status = VL53L1X_GetOffset(0x52, &offset);
103 
104 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
105  if (status != 0) dbcritInt("VL53L1X_GetOffset status = ", status, "");
106  dbwarnInt("VL53L1X_GetOffset = ", offset, " mm");
107 #endif /* DEBUG_DBPRINT */
108 
109  /* Get Xtalk */
110  status = VL53L1X_GetXtalk(0x52, &xtalk);
111 
112 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
113  if (status != 0) dbcritInt("VL53L1X_GetXtalk status = ", status, "");
114  dbwarnInt("VL53L1X_GetXtalk = ", xtalk, " cps");
115 #endif /* DEBUG_DBPRINT */
116 
117  /* Configure distance mode to short (1.3m), better ambient immunity */
118  status = VL53L1X_SetDistanceMode(0x52, 2); /* default = long (4m in the dark with 200 ms timing budget) */
119 
120 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
121 #if DBPRINT_VERBOSE == 1 /* DBPRINT_VERBOSE */
122  if (status != 0) dbcritInt("VL53L1X_SetDistanceMode status = ", status, "");
123 #endif /* DBPRINT_VERBOSE */
124 #endif /* DEBUG_DBPRINT */
125 
126  /* Configure inter-measurement period to 500 ms (IMP, has to be the same or greater than the timing budget) */
127  status = VL53L1X_SetInterMeasurementInMs(0x52, 500); /* default = 100 ms */
128 
129 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
130  if (status != 0) dbcritInt("VL53L1X_SetInterMeasurementInMs status = ", status, "");
131 #endif /* DEBUG_DBPRINT */
132 
133  /* Configure timing budget to 200 ms */
134  status = VL53L1X_SetTimingBudgetInMs(0x52, 200); /* default = 100 ms */
135 
136 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
137  if (status != 0) dbcritInt("VL53L1X_SetTimingBudgetInMs status = ", status, "");
138 #endif /* DEBUG_DBPRINT */
139 
140  /* Get sensor ID */
141  uint16_t id; // must be 0xEEAC ??
142  status = VL53L1X_GetSensorId(0x52, &id);
143 
144 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
145  if (status != 0) dbcritInt("VL53L1X_GetSensorId status = ", status, "");
146  dbinfoInt_hex("VL53L1X_GetSensorId = ", id, "");
147 #endif /* DEBUG_DBPRINT */
148 
149  return (status);
150 }

153 /**************************************************************************//**
154  * @brief
155  * Get a sensor reading.
156  *
157  * @return
158  * The measured distance in mm.
159  *****************************************************************************/
160 uint16_t measureVL53L1X (void)
161 {
162  /* Start ranging */
163  int8_t status = VL53L1X_StartRanging(0x52);
164 
165 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
166  if (status != 0) dbcritInt("VL53L1X_StartRanging status = ", status, "");
167 #endif /* DEBUG_DBPRINT */
168 
169  uint8_t dataReady;
170  uint16_t distance;
171 
172  /* Wait for data ready */
173  while (dataReady != 1)
174  {
175  status = VL53L1X_CheckForDataReady(0x52, &dataReady);
176 
177 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
178  if (status != 0) dbcritInt("VL53L1X_CheckForDataReady status = ", status, "");
179 #endif /* DEBUG_DBPRINT */
180 
181  }
182 
183  /* Get range status */
184  uint8_t state;
185  status = VL53L1X_GetRangeStatus(0x52, &state);
186 
187 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
188  if (status != 0) dbcritInt("VL53L1X_GetRangeStatus status = ", status, "");
189  if (state != 0) dbcritInt("VL53L1X_GetRangeStatus = ", state, "");
190 #endif /* DEBUG_DBPRINT */
191 
192  /* Get distance */
193  status = VL53L1X_GetDistance(0x52, &distance);
194 
195 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
196  if (status != 0) dbcritInt("VL53L1X_GetDistance status = ", status, "");
197 #endif /* DEBUG_DBPRINT */
198 
199  /* Clear interrupt */
200  status = VL53L1X_ClearInterrupt(0x52);
201 
202 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
203  if (status != 0) dbcritInt("VL53L1X_ClearInterrupt status = ", status, "");
204 #endif /* DEBUG_DBPRINT */
205 
206  return (distance);
207 }
