80 /**
81  * @brief This function performs the xtalk calibration.\n
82  * The function returns the xtalk value found and programs the xtalk compensation to the device
83  * @param TargetDistInMm target distance in mm\n
84  * The target distance : the distance where the sensor start to "under range"\n
85  * due to the influence of the photons reflected back from the cover glass becoming strong\n
86  * It's also called inflection point\n
87  * Target reflectance = grey 17%
88  * @return 0: success, !=0: failed
89  * @return xtalk pointer contains the xtalk value found in cps (number of photons in count per second)
90  */
91 int8_t VL53L1X_CalibrateXtalk(uint16_t dev, uint16_t TargetDistInMm, uint16_t *xtalk);
