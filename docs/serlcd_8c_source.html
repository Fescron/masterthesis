63  * Ctrl+q / 17 / 0x11 - Change baud to 57600bps
64  * Ctrl+r / 18 / 0x12 - Change baud to 115200bps
65  * Ctrl+s / 19 / 0x13 - Change baud to 230400bps
66  * Ctrl+t / 20 / 0x14 - Change baud to 460800bps
67  * Ctrl+u / 21 / 0x15 - Change baud to 921600bps
68  * Ctrl+v / 22 / 0x16 - Change baud to 1000000bps
69  * Ctrl+w / 23 / 0x17 - Change baud to 1200bps
70  * Ctrl+x / 24 / 0x18 - Change the contrast. Follow Ctrl+x with number 0 to 255. 120 is default.
71  * Ctrl+y / 25 / 0x19 - Change the TWI address. Follow Ctrl+x with number 0 to 255. 114 (0x72) is default.
72  * Ctrl+z / 26 / 0x1A - Enable/disable ignore RX pin on startup (ignore emergency reset)
73  * '+' / 43 / 0x2B - Set RGB backlight with three following bytes, 0-255
74  * ',' / 44 / 0x2C - Display current firmware version
75  * '-' / 45 / 0x2D - Clear display. Move cursor to home position.
76  * '.' / 46 / 0x2E - Enable system messages (ie, display 'Contrast: 5' when changed)
77  * '/' / 47 / 0x2F - Disable system messages (ie, don't display 'Contrast: 5' when changed)
78  * '0' / 48 / 0x30 - Enable splash screen
79  * '1' / 49 / 0x31 - Disable splash screen

151 /**************************************************************************//**
152  * @brief
153  * Clear the LCD.
154  *
155  * @details
156  * This method also initializes i2C if necessary.
157  *
158  * @return
159  * @return
160  * @li `0` - Byte written correctly.
161  * @li `-1` - `i2cTransferNack`, NACK received during transfer.
162  * @li `-2` - `i2cTransferBusErr`, Bus error during transfer (misplaced START/STOP).
163  * @li `-3` - `i2cTransferArbLost`, Arbitration lost during transfer.
164  * @li `-4` - `i2cTransferUsageFault`, Usage fault.
165  * @li `-5` - `i2cTransferSwFault`, SW fault.
166  * @li `-6` - Coudldn't configure I2C.
167  *****************************************************************************/
168 int8_t clearLCD (void)
169 {
170  /* Initialize I2C if not already the case */
171  if (initI2C() != 0) return (-6);
172 
173  /* Put commands in array */
174  uint8_t data_buf[2] = {SETTING_COMMAND, CLEAR_COMMAND};
175 
176  /* Transfer structure */
177  I2C_TransferSeq_TypeDef i2cTransfer;
178  I2C_TransferReturn_TypeDef result;
179 
180  /* Initializing I2C transfer */
181  i2cTransfer.addr = DISPLAY_ADDRESS << 1; /* Left shift so we go to 8 bit notation instead of 7 */
182  i2cTransfer.flags = I2C_FLAG_WRITE; /* Data written from buf[0].data */
183  i2cTransfer.buf[0].data = data_buf;
184  i2cTransfer.buf[0].len = 2; /* Data is two bytes */
185  result = I2C_TransferInit(BUS_I2C, &i2cTransfer);
186 
187  /* Transmitting I2C data (busy-wait until transfer is complete) */
188  while (result == i2cTransferInProgress) result = I2C_Transfer(BUS_I2C);
189 
190  return (result);
191 }
