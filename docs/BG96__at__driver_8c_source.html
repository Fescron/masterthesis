80  * the BG96 module.

82 void initBG96 (void)
83 {
84  /* Convert ID & Token */
85  charASCII_to_charHex(DEVICE_ID, DEVICE_ID_hex);
86  charASCII_to_charHex(DEVICE_TOKEN, DEVICE_TOKEN_hex);
87 
88 #if PRINT_ID_TOKEN == 1 /* PRINT_ID_TOKEN */
89 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
90  dbprint("INFO: DEVICE_ID = ");
91  dbprintln(DEVICE_ID);
92  dbprint("INFO: DEVICE_ID_hex = ");
93  dbprintln(DEVICE_ID_hex);
94  dbprintln("");
95  dbprint("INFO: DEVICE_TOKEN = ");
96  dbprintln(DEVICE_TOKEN);
97  dbprint("INFO: DEVICE_TOKEN_hex = ");
98  dbprintln(DEVICE_TOKEN_hex);
99  dbprintln("");
100 #endif /* DEBUG_DBPRINT */
101 #endif /* PRINT_ID_TOKEN */
102 
103  /* Enable oscillator to GPIO*/
104  CMU_ClockEnable(cmuClock_GPIO, true);
105 
106  /* Initialize GPIO pin for PWR key */
107  /* In the case of gpioModePushPull, the last argument directly sets the pin state */
108  GPIO_PinModeSet(BG96_PWRKEY_PORT, BG96_PWRKEY_PIN, gpioModePushPull, 0);
109 
110  /*
111  * USART_INITASYNC_DEFAULT:
112  * config.enable = usartEnable // Specifies whether TX and/or RX is enabled when initialization is completed
113  * // (Enable RX/TX when initialization is complete).
114  * config.refFreq = 0 // USART/UART reference clock assumed when configuring baud rate setup
115  * // (0 = Use current configured reference clock for configuring baud rate).
116  * config.baudrate = 115200 // Desired baudrate (115200 bits/s).
117  * config.oversampling = usartOVS16 // Oversampling used (16x oversampling).
118  * config.databits = usartDatabits8 // Number of data bits in frame (8 data bits).
119  * config.parity = usartNoParity // Parity mode to use (no parity).
120  * config.stopbits = usartStopbits1 // Number of stop bits to use (1 stop bit).
121  * config.mvdis = false // Majority Vote Disable for 16x, 8x and 6x oversampling modes (Do not disable majority vote).
122  * config.prsRxEnable = false // Enable USART Rx via PRS (Not USART PRS input mode).
123  * config.prsRxCh = 0 // Select PRS channel for USART Rx. (Only valid if prsRxEnable is true - PRS channel 0).
124  * config.autoCsEnable = false // Auto CS enabling (Auto CS functionality enable/disable switch - disabled).
125  */
126 
127  USART_InitAsync_TypeDef config = USART_INITASYNC_DEFAULT;
128 
129  /* Enable oscillator to USARTx modules */
130  if (BG96_UART == USART0) CMU_ClockEnable(cmuClock_USART0, true);
131  else if (BG96_UART == USART1) CMU_ClockEnable(cmuClock_USART1, true);
132 
133  /* Set pin modes for UART TX and RX pins */
134  if (BG96_UART == USART0)
135  {
136  switch (BG96_UART_LOC)
137  {
138  case 0:
139  GPIO_PinModeSet(gpioPortE, 11, gpioModeInput, 0); /* RX */
140  GPIO_PinModeSet(gpioPortE, 10, gpioModePushPull, 1); /* TX */
141  break;
142  case 2:
143  GPIO_PinModeSet(gpioPortC, 10, gpioModeInput, 0); /* RX */
144  /* No TX pin in this mode */
145  break;
146  case 3:
147  GPIO_PinModeSet(gpioPortE, 12, gpioModeInput, 0); /* RX */
148  GPIO_PinModeSet(gpioPortE, 13, gpioModePushPull, 1); /* TX */
149  break;
150  case 4:
151  GPIO_PinModeSet(gpioPortB, 8, gpioModeInput, 0); /* RX */
152  GPIO_PinModeSet(gpioPortB, 7, gpioModePushPull, 1); /* TX */
153  break;
154  case 5:
155  case 6:
156  GPIO_PinModeSet(gpioPortC, 1, gpioModeInput, 0); /* RX */
157  GPIO_PinModeSet(gpioPortC, 0, gpioModePushPull, 1); /* TX */
158  break;
159  /* default: */
160  /* No default */
161  }
162  }
163  else if (BG96_UART == USART1)
164  {
165  switch (BG96_UART_LOC)
166  {
167  case 0:
168  GPIO_PinModeSet(gpioPortC, 1, gpioModeInput, 0); /* RX */
169  GPIO_PinModeSet(gpioPortC, 0, gpioModePushPull, 1); /* TX */
170  break;
171  case 2:
172  case 3:
173  GPIO_PinModeSet(gpioPortD, 6, gpioModeInput, 0); /* RX */
174  GPIO_PinModeSet(gpioPortD, 7, gpioModePushPull, 1); /* TX */
175  break;
176  case 4:
177  GPIO_PinModeSet(gpioPortA, 0, gpioModeInput, 0); /* RX */
178  GPIO_PinModeSet(gpioPortF, 2, gpioModePushPull, 1); /* TX */
179  break;
180  case 5:
181  GPIO_PinModeSet(gpioPortC, 2, gpioModeInput, 0); /* RX */
182  GPIO_PinModeSet(gpioPortC, 1, gpioModePushPull, 1); /* TX */
183  break;
184  /* default: */
185  /* No default */
186  }
187  }
188 
189  /* Initialize USART asynchronous mode */
190  USART_InitAsync(BG96_UART, &config);
191 
192  /* Route pins */
193  switch (BG96_UART_LOC)
194  {
195  case 0:
196  BG96_UART->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC0;
197  break;
198  case 1:
199  BG96_UART->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC1;
200  break;
201  case 2:
202  BG96_UART->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC2;
203  break;
204  case 3:
205  BG96_UART->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC3;
206  break;
207  case 4:
208  BG96_UART->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC4;
209  break;
210  case 5:
211  BG96_UART->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC5;
212  break;
213  case 6:
214  BG96_UART->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC6;
215  break;
216  default:
217  BG96_UART->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_DEFAULT;
218  }

634 /**************************************************************************//**
