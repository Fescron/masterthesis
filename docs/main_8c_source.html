189 /**************************************************************************//**
190  * @brief
191  * Main function.
192  *****************************************************************************/
193 int main (void)
194 {
195  /* Value to keep the remaining sleep time when waking up using the accelerometer */
196  uint32_t remainingSleeptime = 0;
197 
198  while (1)
199  {
200  switch (MCUstate)
201  {
202  case INIT:
203  {
204  /* Initialize chip */
205  CHIP_Init();
206 
207 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
208  dbprint_INIT(DBG_UART, DBG_UART_LOC, false, false);
209 #endif /* DEBUG_DBPRINT */
210 
211  /* Initialize delay and sleep functionality */
212  delay(0);
213  sleep(2); /* Time so the I2C devices have time to start up */
214 
215  /* Initialize the LCD */
216  clearLCD();
217  writeStringLCD("Initializing...");
218 
219  /* Soft reset SHT30 */ // TODO: SHT30 (0x88) seems to respond to VL53L1X (0x52) commands and break the I2C bus until power cycle
220 // softResetSHT30();
221 // delay(2000);
222 
223  /* Initialize GPIO wake-up */
224  initGPIOwakeup();
225 
226  /* Init VL53L1X */
227  initVL53L1X();
228 
229  /* Init ADC */
230  initADC(CH1);
231 
232  /* Initialize BG96 */
233 #if SENDING_DATA == 1 /* SENDING_DATA */
234  initBG96();
235 #endif /* SENDING_DATA */
236 
237 #if SEND_STATUS_BOOT == 1 /* SEND_STATUS_BOOT */
238 
239 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
240  dbwarn("Sending startup status ...");
241 #endif /* DEBUG_DBPRINT */
242 
243  clearLCD();
244  writeStringLCD("Sending status (0)");
245 
246 #if SENDING_DATA == 1 /* SENDING_DATA */
247  sendStatus(0); /* Send the status data */
248 #endif /* SENDING_DATA */
249 
250 #endif /* SEND_STATUS_BOOT */
251 
252  MCUstate = MEASURE;
253  } break;
254 
255  case MEASURE:
256  {
257  clearLCD();
258  writeStringLCD("Measuring...");
259  displayMeas(); /* Display measurements on following rows */
260 
261  /* Measure and store the temperatures */
262  int32_t temp = readTempDS18B20(0);
263  if (temp < 0) data.t1 = 0;
264  else data.t1 = temp;
265 
266  temp = readTempDS18B20(1);
267  if (temp < 0) data.t2 = 0;
268  else data.t2 = temp;
269 
270 // data.t2 = SHT30_readTemp(); // TODO: SHT30 (0x88) seems to respond to VL53L1X (0x52) commands and break the I2C bus until power cycle
271 
272  /* Measure and store distance */
273  uint16_t distance = measureVL53L1X();
274 
275 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
276  dbinfoInt("distance = ", distance, " mm");
277 #endif /* DEBUG_DBPRINT */
278 
279  data.c = (uint16_t)(((float32_t)(MAX_LENGTH_MM - measureVL53L1X()) / MAX_LENGTH_MM) * 100.0);
280 
281  data.n1 = readADC(CH1);
282 
283  // TODO: measure more things!
284  data.n2 = 20;
285  data.p = 20;
286  data.r = 20;
287 
288 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
289  dbprintln("");
290  dbinfoInt(">> t1: ", data.t1, "");
291  dbinfoInt(">> t2: ", data.t2, "");
292  dbinfoInt(">> c: ", data.c, " %");
293  dbinfoInt(">> n1: ", data.n1, " mA");
294  dbinfoInt(">> n2: ", data.n2, " mA");
295  dbinfoInt(">> p: ", data.p, " mA");
296  dbinfoInt(">> r: ", data.r, " mA");
297  dbprintln("");
298 #endif /* DEBUG_DBPRINT */
299 
300  MCUstate = SEND_DATA;
301  } break;
302 
303  case SEND_DATA:
304  {
305 
306 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
307  dbwarn("Sending measurements ...");
308 #endif /* DEBUG_DBPRINT */
309 
310  clearLCD();
311  writeStringLCD("Sending measurements");
312  displayMeas(); /* Display measurements on following rows */
313 
314 #if SENDING_DATA == 1 /* SENDING_DATA */
315  sendMeasurements(&data);
316 #endif /* SENDING_DATA */
317 
318  MCUstate = SLEEP;
319  } break;
320 
321  case SEND_STATUS:
322  {
323 
324 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
325  dbwarn("Sending STATUS ...");
326 #endif /* DEBUG_DBPRINT */
327 
328  // TODO
329 
330  MCUstate = SLEEP;
331  } break;
332 
333  case SLEEP:
334  {
335  clearLCD();
336  writeStringLCD("Waiting...");
337  displayMeas(); /* Display measurements on following rows */
338 
339 
340  sleep(WAKE_UP_PERIOD_S); /* Go to sleep for xx seconds */
341 
342  MCUstate = WAKEUP;
343  } break;
344 
345  case WAKEUP:
346  {
347  /* Check if we woke up using a button */
348  if (checkBTNinterrupts())
349  {
350 
351 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
352  dbprintln(" ");
353  dbprintln_color("BTN INT", 4);
354 #endif /* DEBUG_DBPRINT */
355 
356  remainingSleeptime = 0; /* Reset passed sleeping time */
357  }
358 
359  /* Check if we woke up using the RTC sleep functionality and act accordingly */
360  if (RTC_checkWakeup())
361  {
362  RTC_clearWakeup(); /* Clear variable */
363 
364 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
365  dbprintln(" ");
366  dbprintln_color("RTC INT", 4);
367 #endif /* DEBUG_DBPRINT */
368 
369  remainingSleeptime = 0; /* Reset passed sleeping time since it's an RTC wakeup */
370  }
371 
372  /* Check if we woke up because of a current pulse */
373  if (PULSE_getTriggered(0) || PULSE_getTriggered(1))
374  {
375  RTC_Enable(false); /* Disable the counter */
376 
377 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
378  dbprintln(" ");
379  dbprintln_color("PULSE INT", 4);
380 #endif /* DEBUG_DBPRINT */
381 
382  clearLCD();
383  writeStringLCD("Sending pulses...");
384  displayMeas(); /* Display measurements on following rows */
385 
386  if (PULSE_getTriggered(0))
387  {
388 
389 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
390  dbwarn("Sending P1 pulses ...");
391 #endif /* DEBUG_DBPRINT */
392 
393 #if SENDING_DATA == 1 /* SENDING_DATA */
394  sendPulse (0, 1);
395 #endif /* SENDING_DATA */
396 
397  PULSE_setTriggered(0, false);
398  }
399  else if (PULSE_getTriggered(1))
400  {
401 
402 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
403  dbwarn("Sending P2 pulses ...");
404 #endif /* DEBUG_DBPRINT */
405 
406 #if SENDING_DATA == 1 /* SENDING_DATA */
407  sendPulse (1, 1);
408 #endif /* SENDING_DATA */
409 
410  PULSE_setTriggered(1, false);
411  }
412 
413  MCUstate = WAKEUP;
414 
415  clearLCD();
416  writeStringLCD("Waiting...");
417  displayMeas(); /* Display measurements on following rows */
418 
419  remainingSleeptime += RTC_getPassedSleeptime(); /* Add the time spend sleeping to the variable */
420 
421  /* Go back to sleep for the remaining time until another measurement (we only take measurements on RTC/button wake-up) */
422  sleep(WAKE_UP_PERIOD_S - remainingSleeptime);
423 
424  }
425  else
426  {
427  MCUstate = MEASURE;
428  }
429 
430  } break;
431 
432  default:
433  {
434  error(10);
435  } break;
436  }
437  }
438 }
